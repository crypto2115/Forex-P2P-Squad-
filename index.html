// ተጠቃሚው መግባቱን መከታተያ ሁኔታ (State) let isLoggedIn = false; let loggedInUserPhone = ""; // 🔄 ገጹ ሲከፈት (Refresh ሲደረግ) የቀድሞ Login መኖሩን ይፈትሻል window.onload = function() { const savedUser = localStorage.getItem('currentUserPhone'); if (savedUser) { isLoggedIn = true; loggedInUserPhone = savedUser; updateAuthHeader(); } }; // የሚያምር የቶስት መልእክት ማሳያ ፋንክሽን function showNotification(text, isError = false) { const toast = document.getElementById('toast'); toast.innerText = text; if(isError) { toast.style.background = "linear-gradient(135deg, #ef4444, #dc2626)"; } else { toast.style.background = "linear-gradient(135deg, #22c55e, #16a34a)"; } toast.classList.add('show'); setTimeout(() => { toast.classList.remove('show'); }, 3000); } // ዌብሳይቱ ላይ የላይኛውን አዝራር የመቀየሪያ ሎጂክ function updateAuthHeader() { const authArea = document.getElementById('auth-area'); if(isLoggedIn) { authArea.innerHTML = ` 👤 ${loggedInUserPhone} `; } else { authArea.innerHTML = ` `; } } // Logout ሲያደርግ መረጃውን ያጠፋል function handleLogout() { isLoggedIn = false; loggedInUserPhone = ""; localStorage.removeItem('currentUserPhone'); // ሴሽኑን ያጠፋል updateAuthHeader(); showNotification("ወጥተዋል!"); } // ወደ ቴሌግራም መረጃ መላኪያ function sendToTelegram(textMessage, successAlert) { const url = `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`; fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ chat_id: CHAT_ID, text: textMessage, parse_mode: "Markdown" }) }) .then(res => { showNotification(successAlert); }) .catch(err => { showNotification("የኔትወርክ ችግር አለ!", true); }); } // Tab Switching function switchTab(tabName) { document.querySelectorAll('.page-section').forEach(page => page.style.display = 'none'); document.querySelectorAll('.nav-tabs .tab').forEach(tab => tab.classList.remove('active')); document.getElementById('page-' + tabName).style.display = 'block'; document.getElementById('tab-' + tabName).classList.add('active'); } function openModal(id) { document.getElementById(id).style.display = 'flex'; } function closeModal(id) { document.getElementById(id).style.display = 'none'; } // Buy/Sell Toggles let currentTradeType = 'buy'; function setTradeType(type) { currentTradeType = type; const buyBtn = document.getElementById('trade-buy-btn'); const sellBtn = document.getElementById('trade-sell-btn'); const submitBtn = document.getElementById('submit-trade-btn'); if(type === 'buy') { buyBtn.style.backgroundColor = '#22c55e'; buyBtn.style.color = 'white'; sellBtn.style.backgroundColor = '#1a202e'; sellBtn.style.color = '#a0aec0'; submitBtn.innerText = "Submit Buy Order"; submitBtn.style.background = "linear-gradient(180deg, #22c55e 0%, #16a34a 100%)"; } else { sellBtn.style.backgroundColor = '#ef4444'; sellBtn.style.color = 'white'; buyBtn.style.backgroundColor = '#1a202e'; buyBtn.style.color = '#a0aec0'; submitBtn.innerText = "Submit Sell Order"; submitBtn.style.background = "linear-gradient(180deg, #ef4444 0%, #dc2626 100%)"; } } // 1. P2P Trade Submit function handleTradeSubmit(event) { event.preventDefault(); if(!isLoggedIn) { showNotification("እባክዎ መጀመሪያ Login ያድርጉ! (ቅድሚያ ይግቡ)", true); openModal('login-modal'); return; } const amount = document.getElementById('trade-amount').value; const method = document.getElementById('payment-method').value; const phone = document.getElementById('user-phone').value; const msg = `🔔 *አዲስ የ P2P ትዕዛዝ ደርሷል!*\n\n` + `👤 *አይነት:* ${currentTradeType.toUpperCase()}\n` + `💰 *መጠን:* ${amount} USDT (BEP20)\n` + `🏦 *ክፍያ መንገድ:* ${method.toUpperCase()}\n` + `📞 *የደንበኛ ስልክ:* ${phone}`; sendToTelegram(msg, "ትዕዛዝዎ በተሳካ ሁኔታ ተመዝግቧል!"); document.getElementById('p2p-form').reset(); } // 2.
